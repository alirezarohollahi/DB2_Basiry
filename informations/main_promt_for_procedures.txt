CREATE PROCEDURE LoadMonthlySnapshot
(
      @YearNumber INT,
      @MonthNumber INT
)
AS
BEGIN

DECLARE @SnapshotDateKey INT
DECLARE @SnapshotDate DATE

SELECT
      @SnapshotDateKey = DateKey,
      @SnapshotDate    = FullDate
FROM DimDate
WHERE PersianYear  = @YearNumber
AND PersianMonth = @MonthNumber
AND IsMonthEnd = 1

DECLARE @PrevMonth INT
DECLARE @PrevYear INT

IF @MonthNumber = 1
BEGIN
      SET @PrevMonth = 12
      SET @PrevYear  = @YearNumber - 1
END
ELSE
BEGIN
      SET @PrevMonth = @MonthNumber - 1
      SET @PrevYear  = @YearNumber
END

--- داده ماه قبل را پیدا و در یک تمپ قرار میدهیم 
SELECT
      PolicyKey,
      CustomerKey,
      InsuranceTypeKey,
      TotalPremiumPaid,
      TotalClaimPaid
INTO #PrevSnapshot
FROM FactPolicyMonthlySnapshot F
JOIN DimDate D
     ON F.SnapshotDateKey = D.DateKey
WHERE D.PersianYear = @PrevYear
AND D.PersianMonth = @PrevMonth
AND D.IsMonthEnd = 1



CREATE TABLE #CurrentMonthMovement
(
      PolicyKey INT,
      PremiumAmount DECIMAL(18,2),
      ClaimAmount DECIMAL(18,2)
)

DECLARE @MonthStartDate DATE
DECLARE @MonthEndDate DATE

SELECT
      @MonthStartDate = MIN(FullDate),
      @MonthEndDate   = MAX(FullDate)
FROM DimDate
WHERE PersianYear  = @YearNumber
AND PersianMonth = @MonthNumber

DECLARE @CurrentDate DATE

SET @CurrentDate =@MonthStartDate


WHILE @CurrentDate <= @MonthEndDate
BEGIN

      INSERT INTO #CurrentMonthMovement

      SELECT
            PolicyKey,
            SUM(PaymentAmount),
            0
      FROM FactPremiumPayment FP
      JOIN DimDate D
           ON FP.DateKey=D.DateKey
      WHERE D.FullDate=@CurrentDate
      GROUP BY PolicyKey

      UNION ALL

      SELECT
            PolicyKey,
            0,
            SUM(ClaimAmount)
      FROM FactClaim FC
      JOIN DimDate D
           ON FC.DateKey=D.DateKey
      WHERE D.FullDate=@CurrentDate
      GROUP BY PolicyKey

      SET @CurrentDate =
      DATEADD(DAY,1,@CurrentDate)

END

-- پیدا کردن جمع یک ماه
SELECT
      PolicyKey,
      SUM(PremiumAmount) AS CurrentPremium,
      SUM(ClaimAmount) AS CurrentClaim
INTO #CurrentMonthAgg
FROM #CurrentMonthMovement
GROUP BY PolicyKey

-- جمع با رکورد ماه قبل
SELECT

      P.PolicyKey,

      ISNULL(S.TotalPremiumPaid,0)
      +
      ISNULL(C.CurrentPremium,0)
      AS TotalPremiumPaid,

      ISNULL(S.TotalClaimPaid,0)
      +
      ISNULL(C.CurrentClaim,0)
      AS TotalClaimPaid

INTO #NewSnapshot

FROM DimPolicy P

LEFT JOIN #PrevSnapshot S
      ON P.PolicyKey=S.PolicyKey

LEFT JOIN #CurrentMonthAgg C
      ON P.PolicyKey=C.PolicyKey

INSERT INTO FactPolicyMonthlySnapshot
(
      SnapshotDate,
      PolicyKey,
      StatusKey,
      TotalPremiumPaid,
      TotalClaimPaid,
      RemainingDaysToExpire
)

SELECT

      @CurrentMonthEnd,

      N.PolicyKey,

      CASE

            WHEN P.ExpireDate < @CurrentMonthEnd
                  THEN 2 -- expired

            WHEN N.TotalClaimPaid > 0
                  THEN 3 -- paid

            ELSE 1 --active

      END,

      N.TotalPremiumPaid,

      N.TotalClaimPaid,

      CASE
            WHEN DATEDIFF(DAY,
                          @CurrentMonthEnd,
                          P.ExpireDate) < 0
                 THEN 0
            ELSE DATEDIFF(DAY,
                          @CurrentMonthEnd,
                          P.ExpireDate)
      END

FROM #NewSnapshot N
JOIN DimPolicy P
     ON N.PolicyKey=P.PolicyKey
end
